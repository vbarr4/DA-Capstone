Temperature

# Original salinity plot
p1 <- combined %>%
  filter(!is.na(Temp), !is.na(Sal), !is.na(season)) %>%
  ggplot(aes(x = Temp, y = Sal)) +
  geom_point(alpha = 0.4, color = "#2C7BB6", size = 1.8) +
  geom_smooth(method = "lm", se = TRUE, color = "#D7191C", linewidth = 1) +
  facet_wrap(~ season) +
  labs(
    title = "Original Salinity",
    x = "Water Temperature (°C)",
    y = "Salinity (PSU)"
  ) +
  theme_minimal(base_size = 13) +
  theme(
    plot.title = element_text(face = "bold", size = 16),
    strip.text = element_text(face = "bold"),
    panel.grid.minor = element_blank()
  )

# Recalculated salinity plot
p2 <- combined %>%
  filter(!is.na(Temp), !is.na(Sal_recalc), !is.na(season)) %>%
  ggplot(aes(x = Temp, y = Sal_recalc)) +

  geom_smooth(method = "lm", se = TRUE, color = "#D7191C", linewidth = 1) +
  facet_wrap(~ season) +
  labs(
    title = "Recalculated Salinity ",
    x = "Water Temperature (°C)",
    y = "Salinity (PSU)"
  ) +
  theme_minimal(base_size = 13) +
  theme(
    plot.title = element_text(face = "bold", size = 16),
    strip.text = element_text(face = "bold"),

  )

Code 
#filter the data to have equal breaks
combined_1 <- combined %>%
  filter(year>2006)

#create lags
combined_1 <- combined %>%
  arrange(DateTimeStamp) %>%
  mutate(
    prcp_1hr  = lag(TotPrcp, 4),
    prcp_3hr  = lag(TotPrcp, 12),
    prcp_6hr  = lag(TotPrcp, 24),
    prcp_12hr = lag(TotPrcp, 48),
    prcp_24hr = lag(TotPrcp, 96)
  )
Code 
combined_1 <- combined_1 %>%
  mutate(
    prcp_3hr_cum = lag(TotPrcp,1) + lag(TotPrcp,2) + lag(TotPrcp,3) + lag(TotPrcp,4) +
                   lag(TotPrcp,5) + lag(TotPrcp,6) + lag(TotPrcp,7) + lag(TotPrcp,8) +
                   lag(TotPrcp,9) + lag(TotPrcp,10) + lag(TotPrcp,11) + lag(TotPrcp,12),

    prcp_24hr_cum = rowSums(sapply(1:96, function(x) lag(TotPrcp, x)), na.rm = FALSE)
  )
Code 
combined_1 %>%
  select(SpCond, prcp_1hr, prcp_3hr, prcp_6hr, prcp_12hr, prcp_24hr) %>%
  pivot_longer(cols = starts_with("prcp_"),
               names_to = "lag",
               values_to = "precip") %>%
  filter(!is.na(SpCond), !is.na(precip)) %>%
  ggplot(aes(x = precip, y = SpCond)) +
  geom_point(alpha = 0.25, color = "#2C7BB6") +
  geom_smooth(method = "lm", color = "#D7191C", se = TRUE) +
  facet_wrap(~ lag, scales = "free_x") +
  labs(
    title = "Effect of Lagged Precipitation on Specific Conductivity",
    x = "Lagged Precipitation",
    y = "Specific Conductivity (mS/cm)"
  ) +
  theme_minimal(base_size = 13) +
  theme(
    plot.title = element_text(face = "bold"),
    strip.text = element_text(face = "bold")
  )
