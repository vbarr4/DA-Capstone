combined <-combined %>% 
  mutate(
  season_astro = case_when(
    date >= as.Date(paste0(year, "-12-21")) | date < as.Date(paste0(year, "-03-20")) ~ "Winter",
    date >= as.Date(paste0(year, "-03-20")) & date < as.Date(paste0(year, "-06-21")) ~ "Spring",
    date >= as.Date(paste0(year, "-06-21")) & date < as.Date(paste0(year, "-09-22")) ~ "Summer",
    TRUE ~ "Fall"
  )
)

#summer dataset by day
summer_daily_avg <- combined_summer %>%
  group_by(date) %>%
  summarise(
    across(
      where(is.numeric),
      ~ mean(.x, na.rm = TRUE)
    ),
    .groups = "drop"
  )

#summer dataset by week
summer_weekly_avg <- combined_summer %>%
  mutate(week = floor_date(date, unit = "week")) %>%
  group_by(week) %>%
  summarise(
    across(
      where(is.numeric),
      ~ mean(.x, na.rm = TRUE)
    ),
    .groups = "drop"
  )

scale_factor <- max(summer_daily_avg$Turb, na.rm = TRUE) / 
                max(summer_daily_avg$TotPrcp, na.rm = TRUE)

ggplot(summer_daily_avg, aes(x = date)) +
  
  # Precipitation (bars)
  geom_col(
    aes(y = TotPrcp * scale_factor, fill = "Precipitation"),
    alpha = 0.6
  ) +
  
  # Turbidity (line)
  geom_line(
    aes(y = Turb, color = "Turbidity"),
    linewidth = 1.1
  ) +
  
  facet_wrap(~ year, scales = "free_x") +
  
  scale_y_continuous(
    name = "Turbidity",
    sec.axis = sec_axis(~ . / scale_factor, name = "Precipitation")
  ) +
  
  # Custom legend colors
  scale_color_manual(
    values = c("Turbidity" = "#2C7FB8")
  ) +
  
  scale_fill_manual(
    values = c("Precipitation" = "#F03B20")
  ) +
  
  labs(
    title = "Turbidity and Precipitation Over Time",
    x = "Date",
    color = "",
    fill = ""
  ) +
  
  scale_x_date(date_labels = "%b %d") +
  
  theme_minimal(base_size = 13) +
  
  theme(
    plot.title = element_text(face = "bold", size = 16),
    strip.text = element_text(face = "bold"),
    panel.grid.minor = element_blank(),
