#select meaningful numeric variables
cor_vars <- combined %>%
  select(
    ATemp, RH, BP, WSpd, MaxWSpd, Wdir, SDWDir, TotPAR, TotPrcp,  #weather
    Temp, DO_mgl, DO_Pct, Sal, SpCond, pH, Turb, ChlFluor, Depth   #water
  ) %>%
  select(where(~sum(!is.na(.)) > 0))#remove NA columns

#compute correlation matrix
cor_matrix <- cor(cor_vars, use = "pairwise.complete.obs")

#remove the upper triangle of the matrix
cor_matrix[upper.tri(cor_matrix)] <- NA

#melt for ggplot
cor_long <- melt(cor_matrix, na.rm = TRUE)
names(cor_long) <- c("Variable1", "Variable2", "Correlation")


#plot lower-triangle matrix
ggplot(cor_long, aes(x = Variable1, y = Variable2, fill = Correlation)) +
  geom_tile(color = "white") +
  scale_fill_gradient2(low = "#a6cee3", mid = "#f7f7f7", high = "#fb9a99", midpoint = 0) +
  geom_text(aes(label = round(Correlation, 2)), size = 3) +
  theme_minimal() +
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1),
    panel.grid = element_blank()
  ) +
  labs(title = "Correlation Matrix", fill = "r")

#detect daily temperature changes 
daily <- combined %>%
  group_by(date) %>%
  summarise(
    mean_ATemp = mean(ATemp, na.rm = TRUE),
    mean_Temp = mean(Temp, na.rm = TRUE),
    total_Prcp = sum(TotPrcp, na.rm = TRUE)
  ) %>%
  mutate(decimal_year = year(date) + (yday(date)-1)/365)

daily <- daily %>%
  arrange(date) %>%
  mutate(
    temp_change = abs(mean_ATemp - lag(mean_ATemp)),
    extreme_temp_change = temp_change > quantile(temp_change, 0.95, na.rm = TRUE)
  )
Code 
#count the extreme temperature changes per month of each year
temp_extremes <- daily %>%
  mutate(
    year = year(date),
    month = month(date)
  ) %>%
  group_by(year, month) %>%
  summarise(
    extreme_changes = sum(extreme_temp_change, na.rm = TRUE),
    n_days = n(),
    rate = extreme_changes / n_days,
    .groups = "drop"
  )
